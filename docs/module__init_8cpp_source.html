40 Status ModuleInit(volatile ModuleInitState* initState, Status (*init)())
41 {
42  for(;;)
43  {
44  if(cpu_CAS(initState, UNINITIALIZED, BUSY))
45  {
46  Status ret = init();
47  *initState = (ret == INFO::OK)? INITIALIZED : ret;
48  COMPILER_FENCE;

50  }
