78  void push_back(const T& item)
79  {
80  if(size_ < n)
81  size_++;
82  // do not complain - overwriting old values is legit
83  // (e.g. sliding window).
84  else
85  head = (head + 1) % n;
86 
87  tail = (tail + 1) % n;
88  data[tail] = item;
89  }

91  void pop_front()
92  {
93  if(size_ != 0)
94  {
95  size_--;
96  head = (head + 1) % n;
97  }
98  else
99  DEBUG_WARN_ERR(ERR::LOGIC); // underflow
100  }
