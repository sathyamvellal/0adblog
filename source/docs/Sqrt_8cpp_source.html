23 u32 isqrt64(u64 n)
24 {
25  u64 op = n;
26  u64 res = 0;
27  u64 one = (u64)1 << 62; // highest power of four <= than the argument
28 
29  while (one > op)
30  one >>= 2;
31 
32  while (one != 0)
33  {
34  if (op >= res + one)
35  {
36  op -= (res + one);
37  res += (one << 1);
38  }
39  res >>= 1;
40  one >>= 2;
41  }
42  return (u32)res;
43 }
